## Terraform出力値確認

```bash
# プロジェクトディレクトリで実行
cd terraform

# 全出力値表示
terraform output

# JSON形式で確認
terraform output -json

# 特定の出力値を取得
terraform output vpc_id
terraform output public_alb_dns_name
terraform output rds_instance_endpoint
terraform output ecs_cluster_name
terraform output nextjs_log_group_name
terraform output go_server_log_group_name

# 出力値を変数に保存
VPC_ID=$(terraform output vpc_id -raw)
ALB_DNS=$(terraform output public_alb_dns_name -raw)
RDS_ENDPOINT=$(terraform output rds_instance_endpoint -raw)
ECS_CLUSTER=$(terraform output ecs_cluster_name -raw)
NEXTJS_LOG_GROUP=$(terraform output nextjs_log_group_name -raw)
GO_LOG_GROUP=$(terraform output go_server_log_group_name -raw)

echo "VPC ID: ${VPC_ID}"
echo "ALB DNS: ${ALB_DNS}"
echo "RDS Endpoint: ${RDS_ENDPOINT}"
```

### ECSタスク確認

```bash
# タスク一覧取得
aws ecs list-tasks \
  --cluster ${ECS_CLUSTER} \

  --cluster ${ECS_CLUSTER} \
  --tasks ${TASK_ARN} \
  --output json | jq '.tasks[0] | {
    taskArn: .taskArn,
    taskDefinitionArn: .taskDefinitionArn,
    lastStatus: .lastStatus,
    desiredStatus: .desiredStatus,
    launchType: .launchType,
    createdAt: .createdAt,
    containers: .containers[] | {
      name: .name,
      lastStatus: .lastStatus,
      exitCode: .exitCode,
      reason: .reason,
      image: .image
    },
    tags: .tags[] | {key: .key, value: .value}

# タスク実行失敗の理由を確認
```

### ターゲット健全性確認（重要）

```bash
# ターゲットグループARN取得
NEXTJS_TG_ARN=$(aws elbv2 describe-target-groups \
  --names ecs-sample-nextjs-tg \
  --query 'TargetGroups[0].TargetGroupArn' \
  --output text)

# ターゲット健全性確認
aws elbv2 describe-target-health \
  --target-group-arn ${NEXTJS_TG_ARN} \
  --query 'TargetHealthDescriptions[*].[Target.Id,Target.Port,TargetHealth.State,TargetHealth.Reason]' \
  --output table

# 健全なターゲット数
aws elbv2 describe-target-health \
  --target-group-arn ${NEXTJS_TG_ARN} \
  --query 'TargetHealthDescriptions[?TargetHealth.State==`healthy`]' \
  --output json | jq 'length'

# 不健全なターゲット確認
aws elbv2 describe-target-health \
  --target-group-arn ${NEXTJS_TG_ARN} \
  --query 'TargetHealthDescriptions[?TargetHealth.State!=`healthy`].[Target.Id,TargetHealth.State,TargetHealth.Description]' \
  --output table
```

## 一括確認スクリプト

### 全リソース確認スクリプト

```bash
#!/bin/bash
# check_all.sh - 全リソース状態確認スクリプト

ENVIRONMENT=${1:-prod}
REGION="ap-northeast-1"

echo "================================================"
echo "ECS/RDS/ALB System Status Check"
echo "Environment: ${ENVIRONMENT}"
echo "Region: ${REGION}"
echo "================================================"

# 環境変数設定
export AWS_REGION=${REGION}
ECS_CLUSTER="ecs-sample-${ENVIRONMENT}"

echo ""
echo "1. ECS Cluster Status"
echo "================================================"
aws ecs describe-clusters --clusters ${ECS_CLUSTER} \
  --query 'clusters[0].[clusterName,status,runningCount,pendingCount]' --output table

echo ""
echo "2. ECS Services Status"
echo "================================================"
aws ecs list-services --cluster ${ECS_CLUSTER} --query 'serviceArns' --output text | while read svc; do
  svc_name=$(basename $svc)
  aws ecs describe-services --cluster ${ECS_CLUSTER} --services $svc \
    --query "services[0].[serviceName,status,desiredCount,runningCount,pendingCount]" --output table
done

echo ""
echo "3. ALB Target Health"
echo "================================================"
aws elbv2 describe-target-groups \
  --query 'TargetGroups[?contains(TargetGroupName, `ecs-sample`)].TargetGroupArn' --output text | while read tg; do
  tg_name=$(echo $tg | rev | cut -d'/' -f1 | rev)
  echo "Target Group: ${tg_name}"
  aws elbv2 describe-target-health --target-group-arn $tg \
    --query 'TargetHealthDescriptions[*].[Target.Id,TargetHealth.State]' --output table
done

echo ""
echo "4. RDS Instance Status"
echo "================================================"
aws rds describe-db-instances --filters "Name=db-instance-id,Values=ecs-sample-*" \
  --query 'DBInstances[0].[DBInstanceIdentifier,DBInstanceStatus,Engine,MultiAZ]' --output table

echo ""
echo "5. CloudWatch Alarms"
echo "================================================"
aws cloudwatch describe-alarms --alarm-name-prefix ecs-sample \
  --query 'MetricAlarms[*].[AlarmName,StateValue,MetricName]' --output table

echo ""
echo "6. Recent Log Events (Last 20)"
echo "================================================"
LOG_GROUP=$(aws logs describe-log-groups --log-group-name-prefix "/ecs/go-server" \
  --query 'logGroups[0].logGroupName' --output text)

if [ ! -z "$LOG_GROUP" ]; then
  aws logs tail ${LOG_GROUP} --max-items 20
fi

echo ""
echo "Check completed at $(date)"
```
